def run(playwright: Playwright) -> None:

    context = browser.new_context(storage_state="D:\\monster_test\\Autotest-monster\\test_case\\UI\\Test_Katana\\cookie_release.json")
    page = context.new_page()

    # ---------------------
    context.close()
    browser.close()


with sync_playwright() as playwright:
